Minesweeper-klipse


(require '[reagent.core :as r])

(def board-width 12)
(def board-height 12)
(def num-mines 18)

(def squares
   (for [x (range board-width)
         y (range board-height)]
     [x y]))

(def mines
  (shuffle
   (into (repeat num-mines 1)
         (repeat (- (count squares) num-mines) 0))))

(def minefield
  (into {} (map vector squares mines)))

(def app-state
  (r/atom
   {:matrix minefield
    :stepped []
    :game-status :in-progress
    :message "Tread lightly..."}))

(defn step! [x y]
  (swap! app-state assoc-in [:stepped]
         (conj (:stepped @app-state) [x y])))

(defn mine-detector [x y]
  (+ ((:matrix @app-state) [(dec x) (dec y)])
     ((:matrix @app-state) [x (dec y)])
     ((:matrix @app-state) [(inc x) (dec y)])
     ((:matrix @app-state) [(inc x) y])
     ((:matrix @app-state) [(inc x) (inc y)])
     ((:matrix @app-state) [x (inc y)])
     ((:matrix @app-state) [(dec x) (inc y)])
     ((:matrix @app-state) [(dec x) y])))

(defn valid-square? [[x y]]
  (and (<= 0 x (dec board-width))
       (<= 0 y (dec board-width))))

(defn filter-squares [[x y]]
  (filter valid-square? (distinct (:stepped @app-state))))

(defn win? []
  (= num-mines
     (-  (* board-height board-width)
         (count (filter-squares (:stepped @app-state))))))

(defn rect-cell [x y]
  [:rect.cell
   {:x (+ 0.05 x) :width 0.9
    :y (+ 0.05 y) :height 0.9
    :fill "white"
    :stroke-width 0.025
    :stroke "black"}])

(defn text-cell [x y]
  [:text
   {:x (+ 0.5 x) :width 1
    :y (+ 0.72 y) :height 1
    :text-anchor "middle"
    :font-size 0.6}
   (if (not= 0 (mine-detector x y))
     (str (mine-detector x y)))])

(defn mine [x y]
  [:g {:stroke "darkred"
       :stroke-width 0.4
       :stroke-linecap "round"
       :transform
       (str "translate(" (+ 0.5 x) "," (+ 0.5 y) ") "
            "scale(0.3)")}
   [:line {:x1 -1 :y1 -1 :x2 1 :y2 1}]
   [:line {:x1 1 :y1 -1 :x2 -1 :y2 1}]])

(defn covered [x y]
  [:rect
   {:width 0.9
    :height 0.9
    :fill "grey"
    :x (+ 0.05 x)
    :y (+ 0.05 y)
    :on-click
    (fn click-square [e]
        (when (= (:game-status @app-state) :in-progress)
        (step! x y)
        (if (win?)
          (do (swap! app-state assoc :game-status :win)
            (swap! app-state assoc :message "Congratulations!")))
        (if (= 1 (minefield [x y]))
          (do (swap! app-state assoc :game-status :dead)
            (swap! app-state assoc :message "Fuck. You blew up.")))))}])

(defn render-board []
  (into
   [:svg.board
    {:view-box (str "0 0 " board-width " " board-height)
     :shape-rendering "auto"
     :style {:max-height "500px"}}]
   (for [x (range board-width)
         y (range board-height)]
     [:g
      [rect-cell x y]
      (if (some #{[x y]} (:stepped @app-state))
        (if (= 1 ((:matrix @app-state) [x y]))
          [mine x y]
          [text-cell x y])
        [covered x y])])))

(defn minesweeper []
  [:center
   [:h1 (:message @app-state)]
   [:button
    {:on-click
     (fn reset-click [e]
       (swap! app-state assoc
              :matrix minefield
              :message "Welcome back"
              :game-status :in-progress
              :stepped []))}
    "Reset"]
   [:div [render-board]]])


[render-board]
