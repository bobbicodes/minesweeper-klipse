Minesweeper-klipse


(require '[reagent.core :as r])

(def board-width 12)
(def board-height 12)
(def num-mines 18)

(def squares
   (for [x (range board-width)
         y (range board-height)]
     [x y]))

(def mines
  (shuffle
   (into (repeat num-mines 1)
         (repeat (- (count squares) num-mines) 0))))

(def minefield
  (into {} (map vector squares mines)))

(def app-state
  (r/atom
   {:matrix minefield
    :stepped []}))

(defn step! [x y]
  (swap! app-state assoc-in [:stepped]
         (conj (:stepped @app-state) [x y])))

(defn rect-cell [x y]
  [:rect.cell
   {:x (+ 0.05 x) :width 0.9
    :y (+ 0.05 y) :height 0.9
    :fill "white"
    :stroke-width 0.025
    :stroke "black"}])

(defn mine [i j]
  [:g {:stroke "darkred"
       :stroke-width 0.4
       :stroke-linecap "round"
       :transform
       (str "translate(" (+ 0.5 i) "," (+ 0.5 j) ") "
            "scale(0.3)")}
   [:line {:x1 -1 :y1 -1 :x2 1 :y2 1}]
   [:line {:x1 1 :y1 -1 :x2 -1 :y2 1}]])

(defn covered [x y]
  [:rect
   {:width 0.9
    :height 0.9
    :fill "grey"
    :x (+ 0.05 x)
    :y (+ 0.05 y)
    :on-click
    (fn blank-click [e]
      (step! x y))}])

(defn render-board []
  (into
   [:svg.board
    {:view-box (str "0 0 " board-width " " board-height)
     :shape-rendering "auto"
     :style {:max-height "500px"}}]
   (for [i (range board-width)
         j (range board-height)]
     [:g
      [rect-cell i j]
      (if (some #{[i j]} (:stepped @app-state))
        (if (= 1 (minefield [i j]))
          [mine i j])
        [covered i j])])))


[render-board]
