Minesweeper-klipse


(require '[reagent.core :as r])

(def board-width 12)
(def board-height 12)
(def num-mines 18)

(def squares
   (for [x (range board-width)
         y (range board-height)]
     [x y]))

(def mines
  (shuffle
   (into (repeat num-mines 1)
         (repeat (- (count squares) num-mines) 0))))

(def minefield
  (into {} (map vector squares mines)))

(defn rect-cell [x y]
  [:rect.cell
   {:x (+ 0.05 x) :width 0.9
    :y (+ 0.05 y) :height 0.9
    :fill "white"
    :stroke-width 0.025
    :stroke "black"}])

(defn render-board []
  (into
   [:svg.board
    {:view-box (str "0 0 " board-width " " board-height)
     :shape-rendering "auto"
     :style {:max-height "500px"}}]
   (for [i (range board-width)
         j (range board-height)]
     [:g
      [rect-cell i j]])))


[render-board]
